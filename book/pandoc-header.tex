\usepackage{graphicx}
\usepackage{geometry}
\usepackage{eso-pic}

% Command to add full-bleed background image
\newcommand\BackgroundPic[1]{%
\put(0,0){%
\parbox[b][\paperheight]{\paperwidth}{%
\vfill
\centering
\includegraphics[width=\paperwidth,height=\paperheight,keepaspectratio=false]{#1}%
\vfill
}}}
